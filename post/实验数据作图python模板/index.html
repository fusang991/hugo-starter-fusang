    save_path: str = None,

    """
    # è®¾ç½®é»˜è®¤é¢œè‰²å’Œæ ‡è®°æ ·å¼
    if colors is None:
        colors = ['blue', 'red', 'green', 'orange', 'purple', 'brown', 'pink', 'gray', 'olive', 'cyan']
    if markers is None:
        markers = ['o', 's', '^', 'D', 'v', '<', '>', 'p', '*', 'h']
    
    # åˆ›å»ºå›¾è¡¨
    plt.figure(figsize=figsize)
    
    for i, dataset in enumerate(datasets):
        # è·å–æ•°æ®
        x = np.array(dataset['x_data'])
        y = np.array(dataset['y_data'])
        label = dataset['label']
        
        # ä½¿ç”¨è¯¥ç»„æ•°æ®çš„è®¾ç½®æˆ–å…¨å±€è®¾ç½®
        method = dataset.get('smooth_method', smooth_method)
        degree = dataset.get('poly_degree', poly_degree)
        
        # é€‰æ‹©é¢œè‰²å’Œæ ‡è®°
        color = colors[i % len(colors)]
        marker = markers[i % len(markers)]
        
        # ç»˜åˆ¶åŸå§‹æ•°æ®ç‚¹
        plt.scatter(x, y, label=f'{label} (åŸå§‹æ•°æ®)', color=color, marker=marker, zorder=5)
        
        # ç”Ÿæˆå¹³æ»‘æ›²çº¿
        x_smooth = np.linspace(x.min(), x.max(), 300)
        
        if method == 'spline':
            # æ£€æŸ¥xè½´æ•°æ®æ˜¯å¦å•è°ƒé€’å¢
            is_monotonic = np.all(np.diff(x) > 0)
            
            if is_monotonic:

                try:

                except:
                    # å¦‚æœæ ·æ¡æ’å€¼å¤±è´¥ï¼Œå›é€€åˆ°å¤šé¡¹å¼æ‹Ÿåˆ
                    print(f"è­¦å‘Š: {label}çš„æ ·æ¡æ’å€¼å¤±è´¥ï¼Œä½¿ç”¨å¤šé¡¹å¼æ‹Ÿåˆä»£æ›¿")
                    coeffs = np.polyfit(x, y, deg=degree)
                    poly_func = np.poly1d(coeffs)
                    y_smooth = poly_func(x_smooth)
                    plt.plot(x_smooth, y_smooth, color=color, linewidth=2, linestyle='--', alpha=0.8)
            else:
                # å¦‚æœæ•°æ®ä¸æ˜¯å•è°ƒé€’å¢çš„ï¼Œä½¿ç”¨UnivariateSpline
                try:
                    from scipy.interpolate import UnivariateSpline
                    # é¦–å…ˆå¯¹xè½´æ•°æ®è¿›è¡Œæ’åº
                    sort_indices = np.argsort(x)
                    x_sorted = x[sort_indices]
                    y_sorted = y[sort_indices]
                    
                    # ä½¿ç”¨UnivariateSplineå¤„ç†éå•è°ƒæ•°æ®
                    spline = UnivariateSpline(x_sorted, y_sorted, s=0)
                    y_smooth = spline(x_smooth)
                    plt.plot(x_smooth, y_smooth, color=color, linewidth=2, linestyle='--', alpha=0.8)
                except:
                    # å¦‚æœUnivariateSplineä¹Ÿå¤±è´¥ï¼Œå›é€€åˆ°å¤šé¡¹å¼æ‹Ÿåˆ
                    print(f"è­¦å‘Š: {label}çš„UnivariateSplineæ’å€¼å¤±è´¥ï¼Œä½¿ç”¨å¤šé¡¹å¼æ‹Ÿåˆä»£æ›¿")
                    coeffs = np.polyfit(x, y, deg=degree)
                    poly_func = np.poly1d(coeffs)
                    y_smooth = poly_func(x_smooth)
                    plt.plot(x_smooth, y_smooth, color=color, linewidth=2, linestyle='--', alpha=0.8)
                    
        elif method == 'polyfit':
            # ä½¿ç”¨å¤šé¡¹å¼æ‹Ÿåˆ
            coeffs = np.polyfit(x, y, deg=degree)
            poly_func = np.poly1d(coeffs)
            y_smooth = poly_func(x_smooth)
            plt.plot(x_smooth, y_smooth, color=color, linewidth=2, linestyle='--', alpha=0.8)
            print(f"{label}çš„å¤šé¡¹å¼æ‹Ÿåˆæ–¹ç¨‹:\n{poly_func}\n")
        elif method == 'univariate_spline':
            # ç›´æ¥ä½¿ç”¨UnivariateSplineæ–¹æ³•
            try:
                from scipy.interpolate import UnivariateSpline
                # é¦–å…ˆå¯¹xè½´æ•°æ®è¿›è¡Œæ’åº
                sort_indices = np.argsort(x)
                x_sorted = x[sort_indices]
                y_sorted = y[sort_indices]
                
                spline = UnivariateSpline(x_sorted, y_sorted, s=0)
                y_smooth = spline(x_smooth)
                plt.plot(x_smooth, y_smooth, color=color, linewidth=2, linestyle='-', alpha=0.8)
            except:
                # å¦‚æœUnivariateSplineå¤±è´¥ï¼Œå›é€€åˆ°å¤šé¡¹å¼æ‹Ÿåˆ
                print(f"è­¦å‘Š: {label}çš„UnivariateSplineæ’å€¼å¤±è´¥ï¼Œä½¿ç”¨å¤šé¡¹å¼æ‹Ÿåˆä»£æ›¿")
                coeffs = np.polyfit(x, y, deg=degree)
                poly_func = np.poly1d(coeffs)
                y_smooth = poly_func(x_smooth)
                plt.plot(x_smooth, y_smooth, color=color, linewidth=2, linestyle='--', alpha=0.8)
    
    # æ·»åŠ å›¾è¡¨å…ƒç´ 
    plt.title(title, fontsize=16, fontweight='bold')
    plt.xlabel(x_label, fontsize=12)
    plt.ylabel(y_label, fontsize=12)
    plt.grid(True, linestyle='--', alpha=0.6)
    plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left')  # å°†å›¾ä¾‹æ”¾åœ¨å›¾è¡¨å¤–éƒ¨
    
    # è°ƒæ•´å¸ƒå±€ï¼Œé˜²æ­¢æ ‡ç­¾è¢«è£åˆ‡

def plot_comparison_chart_non_monotonic(
    datasets: List[Dict],
    x_label: str,
    y_label: str,
    title: str,
    save_path: str = None,
    plot_type: str = 'line'
):
    """
    ç®€åŒ–ç‰ˆçš„å¤šç»„æ•°æ®å¯¹æ¯”å›¾å‡½æ•°ï¼Œé€‚ç”¨äºå¿«é€Ÿå¯¹æ¯”ã€‚
    æ­¤ç‰ˆæœ¬æ”¯æŒéé€’å¢ï¼ˆéå•è°ƒï¼‰çš„xè½´æ•°æ®ã€‚

    å‚æ•°:
    datasets (List[Dict]): æ•°æ®é›†åˆ—è¡¨ï¼Œæ¯ä¸ªå­—å…¸åŒ…å«:
        - 'x_data': xè½´æ•°æ®
        - 'y_data': yè½´æ•°æ®
        - 'label': è¯¥ç»„æ•°æ®çš„æ ‡ç­¾åç§°
    x_label (str): xè½´çš„æ ‡ç­¾åç§°
    y_label (str): yè½´çš„æ ‡ç­¾åç§°
    title (str): å›¾è¡¨çš„æ ‡é¢˜
    save_path (str): å›¾ç‰‡ä¿å­˜è·¯å¾„
    plot_type (str): ç»˜å›¾ç±»å‹ ('line' çº¿å›¾, 'bar' æŸ±çŠ¶å›¾, 'scatter' æ•£ç‚¹å›¾)
    """
    plt.figure(figsize=(10, 6))
    colors = ['blue', 'red', 'green', 'orange', 'purple', 'brown', 'pink', 'gray']
    
    for i, dataset in enumerate(datasets):
        x = np.array(dataset['x_data'])
        y = np.array(dataset['y_data'])
        label = dataset['label']
        color = colors[i % len(colors)]
        
        if plot_type == 'line':
            # å¯¹äºçº¿å›¾ï¼Œå¦‚æœæ•°æ®ä¸æ˜¯å•è°ƒçš„ï¼Œæˆ‘ä»¬éœ€è¦å¯¹æ•°æ®è¿›è¡Œæ’åº
            is_monotonic = np.all(np.diff(x) >= 0)
            if not is_monotonic:
                sort_indices = np.argsort(x)
                x_sorted = x[sort_indices]
                y_sorted = y[sort_indices]
                plt.plot(x_sorted, y_sorted, marker='o', label=label, color=color, linewidth=2)
            else:
                plt.plot(x, y, marker='o', label=label, color=color, linewidth=2)
        elif plot_type == 'bar':
            # æŸ±çŠ¶å›¾éœ€è¦ç‰¹æ®Šå¤„ç†ï¼Œç¡®ä¿xè½´æ•°æ®å¯¹é½
            width = 0.8 / len(datasets)
            x_pos = x + (i - len(datasets)/2 + 0.5) * width
            plt.bar(x_pos, y, width, label=label, color=color, alpha=0.7)
        elif plot_type == 'scatter':
            plt.scatter(x, y, label=label, color=color, s=50, alpha=0.7)
    
    plt.title(title, fontsize=16, fontweight='bold')
    plt.xlabel(x_label, fontsize=12)
    plt.ylabel(y_label, fontsize=12)
    plt.grid(True, linestyle='--', alpha=0.6)
    plt.legend()
    
    if save_path:
        plt.savefig(save_path, dpi=300, bbox_inches='tight')
        print(f"å›¾è¡¨å·²ä¿å­˜è‡³: {save_path}")
    
    plt.show()
